tree("Root")
	parallel
		// as duas arvores de ação acontece ao mesmo tempo
		repeat mute tree("Attack")
		repeat mute tree("Dead")

tree("Dead")
	sequence
		// se tiver menos que 1 de vida
		IsHealthLessThan(0.1)
		// some da cena
		Explode

tree("Attack")
	while SeePlayer
		sequence
			// alveja o player	
			TargetPlayer
			// olha para o player
			LookAtTarget
			// espera um tempo
			Wait(0.1)		
			// atira
			Fire
