tree("Root")
	parallel
		// as duas arvores de ação acontece ao mesmo tempo
		repeat mute tree("Seek")
		repeat mute tree("Dead")

tree("Dead")
	sequence
		// se tiver menos que 1 de vida
		IsHealthLessThan(0.1)
		// some da cena
		Explode

tree("Seek")
	fallback
		// se tiver com metade da vida
		while IsHealthLessThan(90.0)
			fallback
				// começa a segui-lo
				tree("Seguir")

// segue o player
tree("Seguir")
	sequence
		Seeker
